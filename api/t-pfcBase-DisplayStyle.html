Enum pfcDisplayStyle

# include <pfcBase.h>

enum pfcDisplayStyle
{
    pfcDISPSTYLE_DEFAULT,
    pfcDISPSTYLE_WIREFRAME,
    pfcDISPSTYLE_HIDDEN_LINE,
    pfcDISPSTYLE_NO_HIDDEN,
    pfcDISPSTYLE_SHADED,
    pfcDISPSTYLE_FOLLOW_ENVIRONMENT,
    pfcDISPSTYLE_SHADED_WITH_EDGES,
    pfcDisplayStyle_nil
};

Description

This enumerated type represents the the hidden line display options.
